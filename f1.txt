#ADD: MACRO D, A1, A2
	LDAA A1
	ADDA A2
	STAA D
#ENDM

#SUB: MACRO D, A1, A2
	LDAA A1
	SUBA A2
	STAA D
#ENDM

#MULT: MACRO D, A1, P
	LDAA A1
	LDAB P

LOOP	CMPB #0
	BEQ EXIT
	ASLA
	DECB
	BRA LOOP

EXIT	STAA D
#ENDM

#DIV: MACRO D, A1, P
	LDAA A1
	LDAB P

LOOP	CMPB #0
	BEQ EXIT
	ASRA
	DECB
	BRA LOOP

EXIT	STAA D
#ENDM

#REM: MACRO D, A1, P
	LDAA A1
	LDAB P

LOOP	CMPB #0
	BEQ M1
	ASRA
	DECB
	BRA LOOP

M1	LDAB P
	BRA M2

M2	CMPB #0
	BEQ EXIT
	ASLA
	DECB
	BRA M2

EXIT	LDAB A1
	SUBB A
	STAB D
#ENDM
